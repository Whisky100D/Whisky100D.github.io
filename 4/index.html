var CryptoJS = require("crypto-js");
// 只需找到key和iv
var key;
var iv;
function encrypt(data) {
    encryptedKey = CryptoJS.AES.encrypt(data, key, 
    {
        iv: iv,
        mode: CryptoJS.mode.CBC,
        padding: CryptoJS.pad.Pkcs7
    }).toString()
    return encryptedKey
}
